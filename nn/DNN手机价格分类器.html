def create_dataset():
    print('===========================构建张量数据集===========================')
    data = pd.read_csv('手机价格预测.csv')
    x, y = data.iloc[:, :-1].values, data['price_range'].values
    x_train, x_valid, y_train, y_valid = train_test_split(x, y, test_size=0.2, random_state=123)

    scaler = StandardScaler()
    x_train = scaler.fit_transform(x_train)    
    x_valid = scaler.transform(x_valid)       

    train_dataset = TensorDataset(
        torch.tensor(data=x_train, dtype=torch.float32),
        torch.tensor(data=y_train, dtype=torch.int64)
    )
    valid_dataset = TensorDataset(
        torch.tensor(data=x_valid, dtype=torch.float32),
        torch.tensor(data=y_valid, dtype=torch.int64)
    )

    n_features = x.shape[1]
    n_classes = len(np.unique(y))
    print('特征数->', n_features)
    print('类别数->', n_classes)
    return train_dataset, valid_dataset, n_features, n_classes

# 2. 搭建模型
class PhonePriceModel(nn.Module):
    print('===========================搭建神经网络模型===========================')
    def __init__(self, n_features, n_classes):
        super().__init__()
        self.linear1 = nn.Linear(in_features=n_features, out_features=128)
        self.linear2 = nn.Linear(in_features=128, out_features=256)
        self.output = nn.Linear(in_features=256, out_features=n_classes)

    def forward(self, x):
        x = torch.relu(input=self.linear1(x))
        x = torch.relu(input=self.linear2(x))
        output = self.output(x)
        return output

    dataloader = DataLoader(dataset=train_dataset, batch_size=BATCH_SIZE, shuffle=True)

1. 数据处理：切分与标准化

深度学习模型其实非常“单纯”。如果电池容量是 4000，运行内存是 8，模型会误以为“电池”比“内存”重要几百倍。为了消除这种错觉，必须用 StandardScaler 把所有特征拉到同一水平线（平均值为0，标准差为1）。

# 1. 准备数据
def create_dataset():
    print('===========================构建张量数据集===========================')
    # 1.1 加载数据集
    data = pd.read_csv('手机价格预测.csv')
    
    # 1.2 分离特征 x 与 标签 y
    x, y = data.iloc[:, :-1].values, data['price_range'].values
    
    # 1.3 切分：留出 20% 当作期末考试题（验证集），80% 拿来平时练习（训练集）
    x_train, x_valid, y_train, y_valid = train_test_split(x, y, test_size=0.2, random_state=123)

    # 1.4 数据标准化 (消除单位量纲的碾压影响)
    scaler = StandardScaler()
    x_train = scaler.fit_transform(x_train)    # 找规律并转换
    x_valid = scaler.transform(x_valid)        # 严禁在验证集找规律，直接套用训练集的规则！

    # 1.5 包装成 PyTorch 专属的 TensorDataset (张量数据集)
    train_dataset = TensorDataset(
        torch.tensor(data=x_train, dtype=torch.float32),
        torch.tensor(data=y_train, dtype=torch.int64)
    )
    valid_dataset = TensorDataset(
        torch.tensor(data=x_valid, dtype=torch.float32),
        torch.tensor(data=y_valid, dtype=torch.int64) # 保持标签为int64以适配Loss
    )

    n_features = x.shape[1]       # 特征数量 (20个)
    n_classes = len(np.unique(y)) # 类别数量 (4个价格档)
    print('特征数->', n_features)
    print('类别数->', n_classes)
    return train_dataset, valid_dataset, n_features, n_classes

2. 搭建模型：全连接与非线性激活

神经网络就是由一层层的“线性组合（加权求和）”加上“ReLU 激活函数（过滤负数）”堆叠而成的。我们在最后一层不加 Softmax，是因为后面的损失函数会自动帮我们处理。

# 2. 搭建模型
class PhonePriceModel(nn.Module):
    print('===========================搭建神经网络模型===========================')
    def __init__(self, n_features, n_classes):
        super().__init__()
        # 隐藏层1：20 个特征进来，放大到 128 个特征维度
        self.linear1 = nn.Linear(in_features=n_features, out_features=128)
        # 隐藏层2：128 -> 256
        self.linear2 = nn.Linear(in_features=128, out_features=256)
        # 输出层：256 浓缩成 4 个预测类别分数
        self.output = nn.Linear(in_features=256, out_features=n_classes)

    def forward(self, x):
        # 线性运算后，套一层 ReLU 激活函数（赋予网络“拐弯”的非线性能力）
        x = torch.relu(input=self.linear1(x))
        x = torch.relu(input=self.linear2(x))
        # 最后一层直接输出原始得分 (Logits)
        output = self.output(x)
        return output

3. 训练核心：前向、算误差、反向求导、更新步伐

这个模块是深度学习的心脏！每抓取一批数据（BATCH_SIZE），模型就会瞎猜一次，算出有多离谱（loss），反推责任（backward），然后改正参数（step）。

### 3. 模型训练
def train(train_dataset, n_features, n_classes):
    print('===========================模型训练===========================')
    # 1. 创建数据加载器：每次随机抓 32 部手机训练 (shuffle=True 打乱顺序)
    dataloader = DataLoader(dataset=train_dataset, batch_size=BATCH_SIZE, shuffle=True)

    # 2. 实例化模型
    model = PhonePriceModel(n_features=n_features, n_classes=n_classes)

    # 3. 损失函数：多分类必备 CrossEntropyLoss（自带 Softmax 操作）
    criterion = nn.CrossEntropyLoss()

    # 4. 优化器：模型成长的导师 (SGD代表随机梯度下降，LR为学习率步伐大小)
    optimizer = optim.SGD(params=model.parameters(), lr=LR)

    # 5. 开启训练大循环
    epochs = EPOCHS
    for epoch in range(epochs):
        total_loss = 0.0
        start = time.time()
        
        # 内层循环：一批一批地练
        for x, y in dataloader:
            # 5.1 前向传播：模型瞎猜
            y_pred = model(x)
            
            # 5.2 算账：计算预测与真实的差距 (Loss)
            loss = criterion(y_pred, y)
            total_loss += loss.item() * len(x)
            
            # 5.3 反向传播：追究责任
            optimizer.zero_grad() # 每一批次前，把上次算出来的梯度清零！
            loss.backward()       # 核心魔法：算出每个权重该调大还是调小
            
            # 5.4 修正：顺着梯度方向走一步，更新模型参数
            optimizer.step()
            
        dur = time.time() - start
        mean_loss = total_loss / len(train_dataset)
        print(f"epoch: {epoch + 1}, mean_loss: {mean_loss:.3f}, time: {dur:.3f} s")

    # 6. 毕业留念：保存模型权重参数到文件
    torch.save(model.state_dict(), '.price_model.pth')

4. 模型评估：闭卷考试时刻

模型练完了，我们要切回 eval() 模式，用当初藏起来的 20% 验证集数据，看看它真实准确率有多高。

# 4. 模型评估
def evaluate(valid_dataset, n_features, n_classes):
    print('===========================模型评估===========================')
    # 评估时不需要打乱数据 (shuffle=False)
    dataloader = DataLoader(dataset=valid_dataset, batch_size=64, shuffle=False)

    model = PhonePriceModel(n_features=n_features, n_classes=n_classes)
    # 把刚才训练保存好的“脑髓”装进新模型
    model.load_state_dict(torch.load('.price_model.pth'))

    # ★★★ 致命细节：切换到评估模式 ★★★
    # 关闭诸如 Dropout 等只在训练时起作用的作弊管制机制
    model.eval()

    total_correct = 0
    for x, y in dataloader:
        output = model(x) # 得到4个类别的打分
        # dim=-1 代表找最后面那个维度的最大值，返回的是对应索引（0,1,2,3）
        y_pred = torch.argmax(output, dim=-1)   
        # 统计本批次对了几道题
        total_correct += (y_pred == y).sum()
        
    print(f"accuracy: {total_correct / len(valid_dataset):.3f}")
